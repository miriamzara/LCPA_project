\newenvironment{example}[1]
    {%
        \begin{mdframed}[
            leftmargin=20pt,
            rightmargin=20pt,
            linecolor=black, % Color of the vertical line
            linewidth=0.5pt,   % Thickness of the vertical line
            rightline=false, % Disable the right vertical line
            topline=false,   % Disable the top border
            bottomline=false,% Disable the bottom border
            innerleftmargin=10pt, % Space between the line and text
            innerrightmargin=0pt, % No margin on the right
            innertopmargin=3pt, % Top padding
            innerbottommargin=3pt % Bottom padding
        ]%
        \textbf{Example: }#1
        \footnotesize
        \vskip 0.5\baselineskip

    }%
    {%
        \end{mdframed}
        \normalsize
    }

\newenvironment{customProof}
    {
        \footnotesize
        \begin{mdframed}[
            leftmargin=20pt,
            rightmargin=20pt,
            rightline=false,
            leftline=false,
            topline=false,
            bottomline=false
            ]
        \textit{\textbf{Proof:}}
    }
    {
        \hfill$\qed$
        \end{mdframed}
        \normalsize
    }

% Comment up-arrow for equations
\newcommand\uparrowtxt[3][1ex]{%
    \begin{array}[t]{@{}c@{}} 
        #2 \\
        \left\uparrow\vcenter{\hrule height #1}\right.\kern-\nulldelimiterspace\\
        \makebox[0pt]{\scriptsize#3}
    \end{array}%
}

% Comment down-arrow for equations
\newcommand\downarrowtxt[3][1ex]{%
    \begin{array}[b]{@{}c@{}}
        \makebox[0pt]{\scriptsize#3} \\
        \left\downarrow\vcenter{\hrule height #1}\right.\kern-\nulldelimiterspace \\
        #2
   \end{array}%
}
